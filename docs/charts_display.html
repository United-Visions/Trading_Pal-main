Trading Pal Charts Display

Chart Configurations

// Example chart configuration
const chart = new Chart({
  type: 'candlestick',
  data: {
    symbol: 'NASDAQ:AAPL',
    timeframe: '1H',
    indicators: [{
      type: 'ema',
      period: 20,
      color: '#2196f3'
    }, {
      type: 'bollinger',
      period: 20,
      stdDev: 2,
      fill: 'rgba(33, 150, 243, 0.1)'
    }]
  },
  options: {
    theme: 'dark',
    autoScale: true,
    volume: true,
    crosshair: true
  }
});
